//program to generate a pattern as follows
#include <stdio.h>
int main () 
{
   int rows, i, j, n=1;
   printf ("\n\t enter the number of rows:");
   scanf ("%d", &rows);
   for (i=1; i<=rows; i++) 
   {
      for (j=1; j<=i; j++) 
      {
         printf ("%d",n++);
      }
      printf ("\n");
   }
   return 0;
}
